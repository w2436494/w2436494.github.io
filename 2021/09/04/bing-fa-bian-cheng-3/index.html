11、线程池（重点)

面试考点：三大方法、7大参数、4中拒绝策略

池化技术

程序的运行，本质：占用系统的资源！优化资源的使用!=>池化技术

线程池、数据库连接池、对象池///…

线程池化思想：事先准备好一些资源（线程数），当有任务时直接从线程池中获取线程执行，执行完毕后并不销毁，而是将其放入线程池，方便再次使用。线程的创建和销毁很消耗资源。

线程池的优点：

1、降低资源的消耗

2、提高响应速度

3、方便管理

线程复用、可以控制最大的并发数、线程管理

线程池：三大方法 

package juc.pool;

import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;
import java.util.concurrent.ThreadPoolExecutor;

// Executors 工具类 、线程池3大方法
public class Demo01 {
    public static void main(String[] args) {
        // 自定义线程池
//        ExecutorService threadPool = Executors.newSingleThreadExecutor();// 单个线程池的线程池
        ExecutorService threadPool = Executors.newFixedThreadPool(5); // 创建一个固定线程数的线程池
//        ExecutorService threadPool = Executors.newCachedThreadPool(); // 缓存方式创建 可伸缩的，遇强则强，遇弱则弱


        try {
            // 执行10个任务
            for (int i = 0; i < 100; i++) {
                //使用线程池之后，使用线程池来创建线程
                threadPool.execute(()->{
                    System.out.println(Thread.currentThread().getName()+" 执行");
                });
            }
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            // 关闭线程池
            threadPool.shutdownNow();
        }


    }
}

package juc.pool;

import java.util.concurrent.*;

// 线程池7大参数
// 4中拒绝策略
/**
 *  new ThreadPoolExecutor.AbortPolicy() 如果最大线程数和阻塞队列都满了
 *  还有新的任务进来  不处理当前任务  抛出异常
 * new ThreadPoolExecutor.CallerRunsPolicy() // 哪来的去哪里!  这里让main线程去执行'
 * new ThreadPoolExecutor.DiscardPolicy() // 队列满了 丢弃任务 不会抛出异常
 * new ThreadPoolExecutor.DiscardOldestPolicy())
 * 队列满了 去和第一个任务进行竞争,判断第一个任务是否结束,如果是则立马接上.
 * 如果竞争失败 则会丢弃当前任务  不会抛出异常
 */
public class Demo2 {
    public static void main(String[] args) {
        // 线程池最大承载容量 :最大线程数+阻塞队列容量
        ExecutorService threadPool = new ThreadPoolExecutor(2
        ,5,1, TimeUnit.SECONDS,
                new LinkedBlockingQueue<>(3),
                Executors.defaultThreadFactory(),
                // 队列满了 去和第一个任务进行竞争,判断第一个任务是否结束,如果是则立马接上.
                // 如果竞争失败 则会丢弃当前任务  不会抛出异常
                new ThreadPoolExecutor.DiscardOldestPolicy());

        try {
                // 最大承载：Deque + max
                // 超过 RejectedExecutionException
            for (int i = 1; i <= 9; i++) {
                    // 使用了线程池之后，使用线程池来创建线程
                threadPool.execute(()->{
                    System.out.println(Thread.currentThread().getName()+" ok");
                });
            }
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            // 线程池用完，程序结束，关闭线程池
            threadPool.shutdown();
        }
    }
}

线程池的最大线程数如何去设置！
了解：IO密集型，CPU密集型：（调优）

package juc.pool;

import java.util.concurrent.*;

/**
 * 最大线程数如何定义
 */
public class Demo03 {
    public static void main(String[] args) {

        // 自定义线程池
        // 线程池最大承载容量 :最大线程数+阻塞队列容量
        ExecutorService threadPool = new ThreadPoolExecutor(2
                // 最大线程如何定义
                // 1: CPU密集型  几核CPU就可将最大线程数定义为几 ,可以保证CPU的运行效率最高

                // 2: IO密集型:判断程序中十分耗费IO资源的线程数,将最大线程数设置为大于它即可(可以设置为两倍)
                // 假设一个程序有15个大型任务  IO十分占用资源.
                ,Runtime.getRuntime().availableProcessors() // 将最大线程数设置为cpu的核心数
                ,1, TimeUnit.SECONDS,
                new LinkedBlockingQueue<>(3),
                Executors.defaultThreadFactory(),
                new ThreadPoolExecutor.DiscardOldestPolicy());

        try {
            // 最大承载：Deque + max
            // 超过 RejectedExecutionException
            for (int i = 1; i <= 9; i++) {
                // 使用了线程池之后，使用线程池来创建线程
                threadPool.execute(()->{
                    System.out.println(Thread.currentThread().getName()+" ok");
                });
            }
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            // 线程池用完，程序结束，关闭线程池
            threadPool.shutdown();
        }
    }
}

package juc.stream;

import java.util.Arrays;
import java.util.List;

/*	题目要求：一分钟内完成此题，只能用一行代码实现！
*	现在有5个用户！筛选：
*	1、ID 必须是偶数
*	2、年龄必须大于23岁
*	3、用户名转为大写字母
*	4、用户名字母倒着排序
*	5、只输出一个用户！
*/
public class Test {
    public static void main(String[] args) {
        User u1 = new User(1,"a",21);
        User u2 = new User(2,"b",22);
        User u3 = new User(3,"c",23);
        User u4 = new User(4,"d",24);
        User u5 = new User(6,"e",25);

        // 转换为集合存储
        List<User> list = Arrays.asList(u1,u2,u3,u4,u5);

        // 将计算交给流执行
        // lambda 表达式 链式编程 函数式接口  Stream流计算式
        list.stream().filter(u->{return u.getId()%2 == 0;})
                .filter(u -> {return u.getAge() > 23;})
                .map(u -> {return u.getName().toUpperCase();})
                .sorted((uu1,uu2)-> {return uu2.compareTo(uu1);})
                .limit(1)
                .forEach(System.out::println);
    }
}

14、ForkJoin

分支合并

什么是ForkJoin

ForkJoin 在 JDK 1.7 ， 并行执行任务！提高效率。大数据量下使用！

大数据：Map Reduce （把大任务拆分为小任务）

ForkJoin 特点：工作窃取

这个里面维护的都是双端队列。如下图B线程先执行完毕，它会去分担A线程的任务。

ForkJoin

package juc.forkjoin;

import java.util.concurrent.ForkJoinTask;
import java.util.concurrent.RecursiveTask;

/**
 * 求和计算任务
 * 3000 6000(ForkJoin) 9000(Stream并行流)
 * // 如何使用ForkJoin
 * 1、通过ForkJoinPool执行
 * 2、计算任务forkjoinpool.execute(ForkJoinTask task)
 * 3、计算类要基础 ForkJoinTask
 */
public class ForkJoinDemo extends RecursiveTask<Long> {
    private Long start; // 1
    private Long end; // 1990900000

    // 临界值
    private Long temp = 10000L;

    public ForkJoinDemo(Long start, Long end) {
        this.start = start;
        this.end = end;
    }
    // 计算
    @Override
    protected Long compute() {
        if ((end-start) < temp){ // 小于临界值直接计算
            Long sum = 0L;
            for (Long i = start; i <= end ; i++) {
                sum+=i;
            }
            return sum;
        } else { // forkJoin 递归
            Long middle = (end+start)/2; // 中间值
            // 将一个大任务拆成两个任务
            ForkJoinDemo task1 = new ForkJoinDemo(start,middle);
            task1.fork(); // 拆分任务，把任务压入线程队列
            ForkJoinDemo task2 = new ForkJoinDemo(middle+1,end);
            task2.fork(); // 拆分任务，把任务压入线程队列
            return task1.join()+task2.join();
        }
    }
}

package juc.forkjoin;

import java.util.concurrent.ExecutionException;

import java.util.stream.LongStream;

// 3000 6000(ForkJoin) 9000(Stream并行流)
public class Test {
    public static void main(String[] args) throws ExecutionException, InterruptedException {
        test1(); // 674
        test2();
        test3(); // 762
    }

    // 普通程序员
    static void test1(){
        long sum = 0L;
        long start = System.currentTimeMillis();
        for (long i = 0L; i <= 10_0000_0000L; i++) {
            sum+=i;
        }
        long end = System.currentTimeMillis();
        System.out.println("for循环计算 sum="+sum+" 时间："+(end-start));
    }

    // 使用forkjoin
    static void test2() throws ExecutionException, InterruptedException {
        long sum = 0L;
        long start = System.currentTimeMillis();

        ForkJoinPool forkJoinPool = new ForkJoinPool();
        ForkJoinTask<Long> task =  new ForkJoinDemo(0L,10_0000_0000L);
//        forkJoinPool.execute(task); // 执行任务  无返回值
        ForkJoinTask<Long> submit = forkJoinPool.submit(task); // 提交任务 有返回值
        sum = submit.get();
        long end = System.currentTimeMillis();
        System.out.println("forkjoin计算 sum="+sum+" 时间："+(end-start));
    }

    // Stream并行流
    static void test3(){
        long start = System.currentTimeMillis();
        // Stream并行流
        Long sum = LongStream .rangeClosed(0L,10_0000_0000L).parallel().reduce(0,Long::sum);

            return 233; // 可以获得到错误的返回结果

}
